```typescript
private resolveAction(actionName: string): ModelActionConfig | undefined {
    const model = this.model;
    if (!model) return { motion: { group: 'Idle', index: 0 } };
    const capitalized = actionName.charAt(0).toUpperCase() + actionName.slice(1);
    if (model.hasMotionGroup(capitalized)) {
      return { motion: { group: capitalized, index: 0 } };
    }
    if (model.hasMotionGroup(actionName)) {
      return { motion: { group: actionName, index: 0 } };
    }

      return { expression: actionName };
    }
    if (model.hasMotionGroup('Idle')) {
      return { motion: { group: 'Idle', index: 0 } };
    }
    return undefined;
}
```

    3. **resolveAction 函数**:
    ```typescript
    /**
     * 将 action name 解析为具体的 motion 播放指令。
     * 解析链:
     *   1. ACTION_MOTION_MAP 中查找预设映射
     *   2. 如果映射的 group 在模型可用组中 → 返回映射
     *   3. 如果不在，尝试首字母大写匹配模型可用组
     *   4. 如果原始名称（小写）匹配模型可用组
     *   5. Fallback 到 "Idle" 组
     *   6. 如果模型没有 Idle → undefined（调用方静默忽略）
     */
    export function resolveAction(
      actionName: string,
      availableMotionGroups: string[],
    ): ActionMapEntry | null {
      const mapped = ACTION_MOTION_MAP[actionName];
      
      // 1. 尝试预设映射组
      if (mapped && availableMotionGroups.includes(mapped.group)) {
        return mapped;
      }
      
      // 2. 尝试首字母大写匹配
      const capitalized = actionName.charAt(0).toUpperCase() + actionName.slice(1);
      if (availableMotionGroups.includes(capitalized)) {
        return { group: capitalized, index: 0, priority: 2 };
      }
      
      // 3. 尝试原始名称匹配
      if (availableMotionGroups.includes(actionName)) {
        return { group: actionName, index: 0, priority: 2 };
      }
      
      // 4. Fallback 到 Idle
      if (availableMotionGroups.includes('Idle')) {
        return { group: 'Idle', index: 0, priority: 1 };
      }
      
      // 5. 无可用 group → null（调用方静默忽略）
      console.warn(`[Live2D] No motion group found for "${actionName}", no Idle fallback available`);
      return null;
    }
    ```

    ```typescript
    async playAction(actionName: string, options?: PlayActionOptions): Promise<void> {
      if (!this.model) return;
      
      // 清除 idle 自动返回定时器（momentary 动作会重新设置）
      this.clearIdleTimer();
      
      this.currentAction = actionName;
      
      // 通过映射系统解析动作
      const entry = resolveAction(actionName, this.model.motionGroupNames);
      
      if (!entry) {
        console.warn(`[Live2DRenderer] No motion for "${actionName}", skipping`);
        return;
      }
      
      console.log(`[Live2DRenderer] Action: ${actionName} → motion: ${entry.group}[${entry.index ?? 0}]`);
      
      // 播放 motion
      this.model.startMotion(entry.group, entry.index ?? 0, entry.priority ?? 2);
      
      // 处理 momentary 动作自动返回 idle
      const isMomentary = options?.playback === 'momentary' || isMomentaryAction(actionName);
      if (isMomentary) {
        this.scheduleIdle(400);
      }
      
      // per D-17: Phase 1 不做优先级抢占管理，Phase 2 完善
    }
    ```

    ```typescript
    private idleTimerId: number | null = null;
    
    private scheduleIdle(delay: number): void {
      this.clearIdleTimer();
      this.idleTimerId = window.setTimeout(() => {
        this.idleTimerId = null;
        this.playAction('idle');
      }, delay);
    }
    
    private clearIdleTimer(): void {
      if (this.idleTimerId !== null) {
        window.clearTimeout(this.idleTimerId);
        this.idleTimerId = null;
      }
    }
    ```
