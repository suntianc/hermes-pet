    setSpeaking(speaking: boolean, amplitude = 0): void {
      this.isSpeaking = speaking;
      if (speaking) {
        // 振幅削波: 低于阈值的 RMS 视为静音
        this.targetAmplitude = amplitude > this.AMPLITUDE_CLAMP
          ? Math.min(amplitude, 1.0)
          : 0;
      } else {
        this.targetAmplitude = 0;
      }
    }
    ```

    private cacheMouthParamIndex(): void {
      if (!this.userModel) return;
      try {
        const paramId = CubismFramework.getIdManager().getId(CubismDefaultParameterId.MouthOpenY);
        this.cachedMouthParamIndex = this.userModel.getModel().getParameterIndex(paramId);
      } catch {
        this.cachedMouthParamIndex = -1;
      }
    }
    ```

    4. **在 `loadModel()` 末尾添加调用**（在 `this.startLoop()` 之前、`this.playAction('idle')` 之后）：

    5. **添加 `updateLipSync()` 私有方法**（参考 RiveRenderer 的 updateLipSync 模式，但使用 Cubism 的 setParameterValueByIndex）：
    ```typescript
    /** 每帧调用: 平滑振幅并设置 ParamMouthOpenY */
    private updateLipSync(): void {
      // 指数移动平均平滑
      this.currentAmplitude += (this.targetAmplitude - this.currentAmplitude) * this.AMP_LERP_FACTOR;

      if (this.cachedMouthParamIndex < 0) return;
      const model = this.userModel?.getModel();
      if (!model) return;

      let mouthValue: number;

      // 迟滞削波: 防止在阈值附近快速开关导致抖动
      if (this.mouthCurrentlyOpen) {
        // 已开口: 降到 0.03 以下才关闭
        mouthValue = this.currentAmplitude < this.AMPLITUDE_CLAMP - this.AMP_HYSTERESIS
          ? 0
          : this.currentAmplitude;
        this.mouthCurrentlyOpen = mouthValue > 0;
      } else {
        // 已闭口: 超过 0.05 才开口
        mouthValue = this.currentAmplitude > this.AMPLITUDE_CLAMP
          ? this.currentAmplitude
          : 0;
        this.mouthCurrentlyOpen = mouthValue > 0;
      }

      // 仅在变化显著时更新（避免冗余 WASM 调用）
      if (Math.abs(mouthValue - this.lastMouthValue) > 0.005) {
        model.setParameterValueByIndex(this.cachedMouthParamIndex, mouthValue);
        this.lastMouthValue = mouthValue;
      }
    }
    ```

    6. **修改 `startLoop()` 方法**，在渲染循环中每帧调用 `updateLipSync()`：
    ```typescript
    private startLoop(): void {
      const loop = () => {
        if (this.disposed) return;
        if (this.userModel) {
          // 每帧更新唇形同步
          this.updateLipSync();
          this.userModel.getRenderer().drawModel();
        }
        this.animationId = requestAnimationFrame(loop);
      };
      loop();
    }
    ```

    - 迟滞逻辑与 RiveRenderer 保持一致：开口阈值 0.05、闭口阈值 0.03、变化步长 0.005
