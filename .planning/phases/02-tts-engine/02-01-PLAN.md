    ```rust
    fn split_sentences(text: &str) -> Vec<String> {
        let separators = ['。', '！', '？', '；', '\n', '\r', '.', '!', '?', ';'];
        let mut sentences = Vec::new();
        let mut current = String::new();
        for ch in text.chars() {
            current.push(ch);
            if separators.contains(&ch) {
                let trimmed = current.trim().to_string();
                if !trimmed.is_empty() {
                    sentences.push(trimmed);
                }
                current.clear();
            }
        }
        let remaining = current.trim().to_string();
        if !remaining.is_empty() {
            sentences.push(remaining);
        }
        sentences
    }
    ```

    Full algorithm (matching Electron):
    1. `split_sentences()` — split on sentence boundaries
    2. For sentences under `max_chars`, accumulate into current buffer until adding the next would exceed limit
    3. For sentences over `max_chars`, split by comma (Chinese `，` `,` `、`):
       - If sub-parts are still over limit, hard-split at `max_chars`
    4. Return `Vec<TextChunk>`

    Also implement `validate_text(text, max_chars) -> Vec<String>` matching the Electron version: errors if text is empty or exceeds `max_chars * 10`.

    ### FIFO Queue

    ```rust
    use std::collections::VecDeque;
    use crate::tts::config::TTSSpeakOptions;

    #[derive(Debug, Clone)]
    pub struct QueueItem {
        pub id: String,
        pub text: String,
        pub options: Option<TTSSpeakOptions>,
        pub chunks: Vec<TextChunk>,
        pub current_chunk: usize,
    }

    impl QueueItem {
        pub fn new(text: String, options: Option<TTSSpeakOptions>, max_chars: u32) -> Self {
            let chunks = split_text(&text, max_chars);
            Self {
                id: format!("tts_{}", chrono::Utc::now().timestamp_millis()),
                text,
                options,
                chunks,
                current_chunk: 0,
            }
        }

        pub fn is_complete(&self) -> bool {
            self.current_chunk >= self.chunks.len()
        }
    }

    pub struct TtsQueue {
        items: VecDeque<QueueItem>,
        is_processing: bool,
    }

    impl TtsQueue {
        pub fn new() -> Self {
            Self {
                items: VecDeque::new(),
                is_processing: false,
            }
        }

        /// Enqueue a new TTS request. Port of Electron TTSManager.speak() queue push.
        pub fn enqueue(&mut self, text: String, options: Option<TTSSpeakOptions>, max_chars: u32) -> Option<String> {
            let item = QueueItem::new(text, options, max_chars);
            let id = item.id.clone();
            self.items.push_back(item);
            Some(id)
        }

        /// Dequeue the next item (FIFO)
        pub fn dequeue(&mut self) -> Option<QueueItem> {
            self.items.pop_front()
        }

        /// Peek at the next item without removing
        pub fn peek(&self) -> Option<&QueueItem> {
            self.items.front()
        }

        /// Clear all pending items
        pub fn clear(&mut self) {
            self.items.clear();
        }

        pub fn len(&self) -> usize {
            self.items.len()
        }

        pub fn is_empty(&self) -> bool {
            self.items.is_empty()
        }

        pub fn is_processing(&self) -> bool {
            self.is_processing
        }

        pub fn set_processing(&mut self, processing: bool) {
            self.is_processing = processing;
        }
    }
    ```
